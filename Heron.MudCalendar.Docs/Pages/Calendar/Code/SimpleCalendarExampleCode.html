            <MudChip Label="true" Color="@context.Color" Style="width: 100%; height: 100%; margin: 0; min-height: 36px;">@context.Title</MudChip>

@code {
    private List<CalendarItem<Event>> _events = new();
    
    protected override void OnInitialized()
    {
        base.OnInitialized();
        
        _events = new List<CalendarItem<Event>>
        {
            new CalendarItem<Event>
            {
                Start = DateTime.Today.AddHours(10), End = DateTime.Today.AddHours(11),
                Content = new Event { Title = "Event today", Color = Color.Secondary }
            },
            new CalendarItem<Event>
            {
                Start = DateTime.Today.AddDays(1).AddHours(11), End = DateTime.Today.AddDays(1).AddHours(12.5),
                Content = new Event { Title = "Event tomorrow", Color = Color.Tertiary }
            }
        };
    }
    
    private class Event
    {
        public string Title { get; set; } = string.Empty;

        public Color Color { get; set; } = Color.Primary;
    }
}
